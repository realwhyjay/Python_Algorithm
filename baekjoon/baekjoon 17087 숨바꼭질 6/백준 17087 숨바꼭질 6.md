## 최종 답안 

```python
def GCD(a, b):
    if a < b:
        a, b = b, a
    if b == 0:
        return a
    else:
        return GCD(b, a % b)


n, s = map(int, (input().split()))
location = list(map(int, (input().split())))

# 값을 구하기위해 현재위치에서 동생들과의 위치의 차를 구한다
# 동생의 위치가 내 위치보다 앞일수도, 뒤일수도 있기 때문에 절대값을 구하는 함수인 abs()를 사용해주었다
for i in range(len(location)):
    location[i] = abs(s - location[i])

GCDresult = location[0]

# 여러 수의 최대공약수를 구하는 방법은, 
# 두 수의 최대공약수를 구하고 이 값과 다른 수의 최대공약수를 구하는식으로 반복하면 된다! 
for j in range(len(location)):
    GCDresult = GCD(location[j], GCDresult)

print(GCDresult)

```
